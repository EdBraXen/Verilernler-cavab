public static String correctSolution(int[] arr) {
    long total = 0;
    long min = Long.MAX_VALUE;
    long max = Long.MIN_VALUE;

    for (int v : arr) {
        total += v;
        if (v < min) min = v;
        if (v > max) max = v;
    }

    long minSum = total - max;
    long maxSum = total - min;

    return minSum + " " + maxSum;
}

public static String correctSolution(String s, int k) {

    Map<Integer, Integer> count = new HashMap<>();

        count.put(x, count.getOrDefault(x, 0) + 1);
    }

    List<Integer> duplicates = new ArrayList<>();
    for (Map.Entry<Integer, Integer> e : count.entrySet()) {
        if (e.getValue() > 1) {
            duplicates.add(e.getKey());
        }
    }

    Collections.sort(duplicates);

    for (int i = 0; i < duplicates.size(); i++) {

    }

public static int correctSolution(int[][] tree) {
    // tree[i] = {leftChild, rightChild}, index 1-based
    boolean[] isRoot = new boolean[tree.length];
    Arrays.fill(isRoot, true);

    // Identify the root (node that is never a child)
    for (int i = 1; i < tree.length; i++) {
        int left = tree[i][0];
        int right = tree[i][1];
        if (left != -1) isRoot[left] = false;
        if (right != -1) isRoot[right] = false;
    }

    int root = 1;
    for (int i = 1; i < isRoot.length; i++) {
        if (isRoot[i]) {
            root = i;
            break;
        }
    }

    return height(root, tree);
}

private static int height(int node, int[][] tree) {
    if (node == -1) return -1;
    int left = tree[node][0];
    int right = tree[node][1];
    return Math.max(height(left, tree), height(right, tree)) + 1;
}

public static long correctSolution(long[] price) {
    int n = price.length;
    TreeMap<Long, Integer> map = new TreeMap<>();
    long minLoss = Long.MAX_VALUE;

    for (int i = 0; i < n; i++) {
        Long higher = map.higherKey(price[i]);

public static int correctSolution(String s) {
    int n = s.length();
    int ans = 0;
    Map<Character, Integer> map = new HashMap<>();
    for (int j = 0, i = 0; j < n; j++) {
        if (map.containsKey(s.charAt(j))) {
            i = Math.max(map.get(s.charAt(j)), i);

        ans = Math.max(ans, j - i + 1);
        map.put(s.charAt(j), j + 1);
    }

    left = 0; right = n - 1;
    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (arr[mid] <= x) left = mid + 1;
        else right = mid - 1;
    }
    if (right >= 0 && arr[right] == x) last = right;

    return new int[]{first, last};
}

public static List<List<String>> solveNQueens(int n) {
    List<List<String>> res = new ArrayList<>();
    int[] board = new int[n];

    return res;

private static void placeQueens(int row, int[] board, List<List<String>> res, int n) {
    if (row == n) {
        List<String> solution = new ArrayList<>();
        for (int r = 0; r < n; r++) {
            char[] line = new char[n];
            Arrays.fill(line, '.');
            line[board[r]] = 'Q';
            solution.add(new String(line));
        }
        res.add(solution);
        return;
    }
    for (int col = 0; col < n; col++) {
        if (isSafe(board, row, col)) {
            board[row] = col;
            placeQueens(row + 1, board, res, n);
        }
    }
}

private static boolean isSafe(int[] board, int row, int col) {
    for (int i = 0; i < row; i++) {
        if (board[i] == col || Math.abs(board[i] - col) == row - i)
            return false;
    }
    return true;
}

public static List<Integer> spiralOrder(int[][] matrix) {
    List<Integer> result = new ArrayList<>();
    int top = 0, bottom = matrix.length - 1, left = 0, right = matrix[0].length - 1;

    while (top <= bottom && left <= right) {
        // Move from left to right along the top row
        for (int i = left; i <= right; i++) {
            result.add(matrix[top][i]);
        }
        top++;

        for (int i = top; i <= bottom; i++) {
            result.add(matrix[i][right]);
        }
        right--;

        // Move from right to left along the bottom row (if there are remaining rows)
        if (top <= bottom) {
            for (int i = right; i >= left; i--) {
                result.add(matrix[bottom][i]);
            }
            bottom--;
        }

        // Move from bottom to top along the left column (if there are remaining columns)
        if (left <= right) {
            for (int i = bottom; i >= top; i--) {
                result.add(matrix[i][left]);
            }
            left++;
        }
    }

    return result;
}

public static boolean correctSolution(ListNode head) {
    if (head == null) return false;

    ListNode slow = head;
    ListNode fast = head;

    while (fast != null && fast.next != null) {
        slow = slow.next;
        fast = fast.next.next;

}

public static List<Integer> correctSolution(int[][] seats) {  
    int rows = seats.length;  
    int cols = seats[0].length;  
    List<Integer> blockedSeats = new ArrayList<>();  

    for (int col = 0; col < cols; col++) {  
        for (int row = 1; row < rows; row++) {  
            if (seats[row][col] < seats[row - 1][col]) {  
                blockedSeats.add(row * cols + col + 1);  
            }  
        }  
    }  

    return blockedSeats;  
}

public static String correctSolution(String s) {  
    String result = "";  
    int n = s.length();  
    int count = 1;  

    for (int i = 1; i < n; i++) {  
        if (s.charAt(i) == s.charAt(i - 1)) {  
            count++;  
        } else {  
            result += s.charAt(i - 1) + Integer.toString(count);  
            count = 1;
        }  
    }  

    result += s.charAt(n - 1) + Integer.toString(count);  

    return result.length() < n ? result : s;  
}
